73static inline pio_sm_config quadrature_encoder_program_get_default_config(uint offset) {
74 pio_sm_config c = pio_get_default_sm_config();
75 sm_config_set_wrap(&c, offset + quadrature_encoder_wrap_target, offset + quadrature_encoder_wrap);
76 return c;
77}

82static inline void quadrature_encoder_program_init(PIO pio, uint sm, uint offset, uint pin, int max_step_rate){
83 pio_sm_set_consecutive_pindirs(pio, sm, pin, 2, false);
84 gpio_pull_up(pin);
85 gpio_pull_up(pin + 1);
86 pio_sm_config c = quadrature_encoder_program_get_default_config(offset);
87 sm_config_set_in_pins(&c, pin);
88 sm_config_set_jmp_pin(&c, pin);
89 sm_config_set_in_shift(&c, false, false, 32);
90 sm_config_set_fifo_join(&c, PIO_FIFO_JOIN_NONE);
91 if (max_step_rate == 0) {
92 sm_config_set_clkdiv(&c, 1.0);
93 } else {
94 float div = (float)clock_get_hz(clk_sys) / (14 * max_step_rate);
95 sm_config_set_clkdiv(&c, div);
96 }
97 pio_sm_init(pio, sm, offset, &c);
98 pio_sm_set_enabled(pio, sm, true);
99}

101static inline void quadrature_encoder_request_count(PIO pio, uint sm){
102 pio->txf[sm] = 1;

105static inline int32_t quadrature_encoder_fetch_count(PIO pio, uint sm){
106 while (pio_sm_is_rx_fifo_empty(pio, sm))
107 tight_loop_contents();
108 return pio->rxf[sm];
109}

111static inline int32_t quadrature_encoder_get_count(PIO pio, uint sm){
112 quadrature_encoder_request_count(pio, sm);
113 return quadrature_encoder_fetch_count(pio, sm);
114}
