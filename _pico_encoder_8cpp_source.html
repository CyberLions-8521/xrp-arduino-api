28// initialize the PIO state and the substep_state_t structure that keeps track
29// of the encoder state
30static inline void pico_encoder_program_init(PIO pio, uint sm, uint pin_A)
31{
32 uint pin_state, position, ints;
33
34 pio_gpio_init(pio, pin_A);
35 pio_gpio_init(pio, pin_A + 1);
36
37 pio_sm_set_consecutive_pindirs(pio, sm, pin_A, 2, false);
38
39 pio_sm_config c = pico_encoder_program_get_default_config(0);
40 sm_config_set_in_pins(&c, pin_A); // for WAIT, IN
41 // shift to left, auto-push at 32 bits
42 sm_config_set_in_shift(&c, false, true, 32);
43 sm_config_set_out_shift(&c, true, false, 32);
44 // don't join FIFO's
45 sm_config_set_fifo_join(&c, PIO_FIFO_JOIN_NONE);
46
47 // always run at sysclk, to have the maximum possible time resolution
48 sm_config_set_clkdiv(&c, 1.0);
49
50 pio_sm_init(pio, sm, 0, &c);
51
52 // set up status to be rx_fifo < 1
53 pio->sm[sm].execctrl = ((pio->sm[sm].execctrl & 0xFFFFFF80) | 0x12);
54
55 // init the state machine according to the current phase. Since we are
56 // setting the state running PIO instructions from C state, the encoder may
57 // step during this initialization. This should not be a problem though,
58 // because as long as it is just one step, the state machine will update
59 // correctly when it starts. We disable interrupts anyway, to be safe
60 ints = save_and_disable_interrupts();
61
62 pin_state = (gpio_get_all() >> pin_A) & 3;
63
64 // to setup the state machine, we need to set the lower 2 bits of OSR to be
65 // the negated pin state
66 pio_sm_exec(pio, sm, pio_encode_set(pio_y, ~pin_state));
67 pio_sm_exec(pio, sm, pio_encode_mov(pio_osr, pio_y));
68
69 // also set the Y (current step) so that the lower 2 bits of Y have a 1:1
70 // mapping to the current phase (input pin state). That simplifies the code
71 // to compensate for differences in encoder phase sizes:
72 switch (pin_state) {
73 case 0: position = 0; break;
74 case 1: position = 3; break;
75 case 2: position = 1; break;
76 case 3: position = 2; break;
77 }
78 pio_sm_exec(pio, sm, pio_encode_set(pio_y, position));
79
80 pio_sm_set_enabled(pio, sm, true);
81
82 restore_interrupts(ints);
83}

85static inline void pico_encoder_get_counts(PIO pio, uint sm, uint *step, int *cycles, uint *us)
86{
87 int i, pairs;
88 uint ints;
89
90 pairs = pio_sm_get_rx_fifo_level(pio, sm) >> 1;
91
92 // read all data with interrupts disabled, so that there can not be a
93 // big time gap between reading the PIO data and the current us
94 ints = save_and_disable_interrupts();
95 for (i = 0; i < pairs + 1; i++) {
96 *cycles = pio_sm_get_blocking(pio, sm);
97 *step = pio_sm_get_blocking(pio, sm);
98 }
99 *us = time_us_32();
100 restore_interrupts(ints);
101}

120void PicoEncoder::setPhases(int phases)
121{
122 calibration_data[0] = 0;
123 calibration_data[1] = (phases & 0xFF);
124 calibration_data[2] = calibration_data[1] + ((phases >> 8) & 0xFF);
125 calibration_data[3] = calibration_data[2] + ((phases >> 16) & 0xFF);
126}

130void PicoEncoder::read_pio_data(uint *step, uint *step_us, uint *transition_us, int *forward)
131{
132 int cycles;
133
134 // get the raw data from the PIO state machine
135 pico_encoder_get_counts(pio, sm, step, &cycles, step_us);
136
137 // when the PIO program detects a transition, it sets cycles to either zero
138 // (when step is incrementing) or 2^31 (when step is decrementing) and keeps
139 // decrementing it on each 13 clock loop. We can use this information to get
140 // the time and direction of the last transition
141 if (cycles < 0) {
142 cycles = -cycles;
143 *forward = 1;
144 } else {
145 cycles = 0x80000000 - cycles;
146 *forward = 0;
147 }
148 *transition_us = *step_us - ((cycles * 13) / clocks_per_us);
149}

151// get the sub-step position of the start of a step
152uint PicoEncoder::get_step_start_transition_pos(uint step)
153{
154 return ((step << 6) & 0xFFFFFF00) | calibration_data[step & 3];
155}

157// compute speed in "sub-steps per 2^20 us" from a delta substep position and
158// delta time in microseconds
159static int substep_calc_speed(int delta_substep, int delta_us)
160{
161 return ((int64_t) delta_substep << 20) / delta_us;
162}

165// function to measure the difference between the different steps on the encoder
166int PicoEncoder::measurePhases(void)
167{
168 int forward, s1, s2, s3;
169 uint count, cur_us, last_us, step_us, step, last_step, start_us, delta;
170 int64_t sum[4], total;
171
172 memset(sum, 0, sizeof(sum));
173
174 // keep reading the PIO state in a tight loop to get all steps and use the
175 // transition measures of the PIO code to measure the time of each step
176 last_step = 0;
177 last_us = 0;
178 count = 0;
179
180 start_us = time_us_32();
181
182 while (1) {
183
184 read_pio_data(&step, &step_us, &cur_us, &forward);
185
186 // if we don't have a transition, just check some stopping conditions and
187 // keep waiting for a transition
188 if (step == last_step) {
189 delta = time_us_32() - start_us;
190 // if we have less than 20 steps per second, this is too slow for the
191 // calibration to produce good results
192 if (count * (1000000 / 20) + 500000 < delta)
193 return -1;
194
195 // never take more than 10 seconds to run the calibration, even if we only
196 // have 200 steps. Just use whatever information was gathered so far
197 if (delta > 10000000)
198 break;
199
200 // if we already have 1024 steps and have measured for over 2 seconds, we
201 // are fine as well. If we have a really fast step rate, then in two
202 // seconds we may be able to get a high step count
203 if (count > 1024 && delta > 2000000)
204 break;
205
206 continue;
207 }
208
209 // check that we are not skipping steps
210 if (count > 10 && abs((int)(last_step - step)) > 1)
211 return -2;
212
213 // sum the step period in the correct step sum
214 if (last_us != 0) {
215 if (forward)
216 sum[(step - 1) & 3] += cur_us - last_us;
217 else
218 sum[(step + 1) & 3] += cur_us - last_us;
219 }
220
221 last_step = step;
222 last_us = cur_us;
223 count++;
224 }
225
226 // scale the sizes to a total of 256 to be used as sub-steps
227 total = sum[0] + sum[1] + sum[2] + sum[3];
228 s1 = (sum[0] * 256 + total / 2) / total;
229 s2 = ((sum[0] + sum[1]) * 256 + total / 2) / total;
230 s3 = ((sum[0] + sum[1] + sum[2]) * 256 + total / 2) / total;
231
232 s3 -= s2;
233 s2 -= s1;
234
235 return s1 | (s2 << 8) | (s3 << 16);
236}

248// try to claim all SM's and load the program to PIO 'pio'. Return true on
249// success, false on failure
250static bool pico_encoder_claim_pio(PIO pio)
251{
252 // check that we can load the program on this PIO
253 if (!pio_can_add_program(pio, &pico_encoder_program))
254 return false;
255
256 // check that all SM's are free. Some libraries claim an SM and later load the
257 // PIO code and that would not interact well with code that uses the entire
258 // PIO code space. So just make sure we can claim all the SM's to prevent this
259 for (int i = 0; i < 4; i++)
260 if (pio_sm_is_claimed(pio, i))
261 return false;
262
263 // claim all SM's
264 for (int i = 0; i < 4; i++)
265 pio_sm_claim(pio, i);
266
267 // load the code into the PIO
268 pio_add_program(pio, &pico_encoder_program);
269
270 return true;
271}

273int PicoEncoder::begin(int firstPin, uint resolution, bool pullUp)
274{
275 _resolution = resolution;
276 int forward, gpio_pin;
277
278 // the first encoder needs to load a PIO with the PIO code
279 if (encoder_count == 0) {
280 // it can either use pio0
281 if (pico_encoder_claim_pio(pio0))
282 pio_used[0] = pio0;
283 else if (pico_encoder_claim_pio(pio1))
284 pio_used[0] = pio1; // or pio1
285 else
286 return -1; // or give up
287
288 } else if (encoder_count == 4) {
289 // the 5th encoder needs to try to use the other PIO
290 pio_used[1] = (pio_used[0] == pio0) ? pio1 : pio0;
291 if (!pico_encoder_claim_pio(pio_used[1]))
292 return -1;
293
294 } else if (encoder_count >= 8) {
295 // we don't support more than 8 encoders
296 return -2;
297 }
298
299 // assign the PIO and state machine and update the encoder count
300 pio = pio_used[encoder_count / 4];
301 sm = encoder_count % 4;
302 encoder_count++;
303
304 // set all fields to zero by default
305 prev_trans_pos = 0;
306 prev_low = 0;
307 prev_high = 0;
308 idle_stop_sample_count = 0;
309 speed_2_20 = 0;
310 speed = 0;
311
312 // save the pin used
313 gpio_pin = translate_pin(firstPin);
314
315 // the PIO init code sets the pins as inputs. Optionally turn on pull ups
316 // here if the user asked for it
317 if (pullUp) {
318 gpio_set_pulls(gpio_pin, true, false);
319 gpio_set_pulls(gpio_pin + 1, true, false);
320 }
321
322 // initialize the PIO program (and save the PIO reference)
323 pico_encoder_program_init(pio, sm, gpio_pin);
324
325 // start "stopped" so that we don't use stale data to compute speeds
326 stopped = 1;
327
328 // cache the PIO cycles per us
329 clocks_per_us = (clock_get_hz(clk_sys) + 500000) / 1000000;
330
331 // initialize the "previous state"
332 read_pio_data(&step, &prev_step_us, &prev_trans_us, &forward);
333
334 position = get_step_start_transition_pos(step) + 32;
335
336 return 0;
337}

340// read the PIO data and update the speed / position estimate
341void PicoEncoder::update(void)
342{
343 uint new_step, step_us, transition_us, transition_pos, low, high;
344 int forward, speed_high, speed_low;
345
346 // read the current encoder state from the PIO
347 read_pio_data(&new_step, &step_us, &transition_us, &forward);
348
349 // from the current step we can get the low and high boundaries in
350 // substeps of the current position
351 low = get_step_start_transition_pos(new_step);
352 high = get_step_start_transition_pos(new_step + 1);
353
354 // if we were not stopped, but the last transition was more than
355 // "idle_stop_samples" ago, we are stopped now
356 if (new_step == step)
357 idle_stop_sample_count++;
358 else
359 idle_stop_sample_count = 0;
360
361 if (!stopped && idle_stop_sample_count >= idle_stop_samples) {
362 speed = 0;
363 speed_2_20 = 0;
364 stopped = 1;
365 }
366
367 // when we are at a different step now, there is certainly a transition
368 if (step != new_step) {
369 // the transition position depends on the direction of the move
370 transition_pos = forward ? low : high;
371
372 // if we are not stopped, that means there is valid previous transition
373 // we can use to estimate the current speed
374 if (!stopped)
375 speed_2_20 = substep_calc_speed(transition_pos - prev_trans_pos, transition_us - prev_trans_us);
376
377 // if we have a transition, we are not stopped now
378 stopped = 0;
379 // save the timestamp and position of this transition to use later to
380 // estimate speed
381 prev_trans_pos = transition_pos;
382 prev_trans_us = transition_us;
383 }
384
385 // if we are stopped, speed is zero and the position estimate remains
386 // constant. If we are not stopped, we have to update the position and speed
387 if (!stopped) {
388 // although the current step doesn't give us a precise position, it does
389 // give boundaries to the position, which together with the last
390 // transition gives us boundaries for the speed value. This can be very
391 // useful especially in two situations:
392 // - we have been stopped for a while and start moving quickly: although
393 // we only have one transition initially, the number of steps we moved
394 // can already give a non-zero speed estimate
395 // - we were moving but then stop: without any extra logic we would just
396 // keep the last speed for a while, but we know from the step
397 // boundaries that the speed must be decreasing
398
399 // if there is a transition between the last sample and now, and that
400 // transition is closer to now than the previous sample time, we should
401 // use the slopes from the last sample to the transition as these will
402 // have less numerical issues
403 if (prev_trans_us > prev_step_us &&
404 (int)(prev_trans_us - prev_step_us) > (int)(step_us - prev_trans_us)) {
405 speed_high = substep_calc_speed(prev_trans_pos - prev_low, prev_trans_us - prev_step_us);
406 speed_low = substep_calc_speed(prev_trans_pos - prev_high, prev_trans_us - prev_step_us);
407 } else {
408 // otherwise use the slopes from the last transition to now
409 speed_high = substep_calc_speed(high - prev_trans_pos, step_us - prev_trans_us);
410 speed_low = substep_calc_speed(low - prev_trans_pos, step_us - prev_trans_us);
411 }
412 // make sure the current speed estimate is between the maximum and
413 // minimum values obtained from the step slopes
414 if (speed_2_20 > speed_high)
415 speed_2_20 = speed_high;
416 if (speed_2_20 < speed_low)
417 speed_2_20 = speed_low;
418
419 // convert the speed units from "sub-steps per 2^20 us" to "sub-steps
420 // per second"
421 speed = (speed_2_20 * 62500LL) >> 16;
422
423 // estimate the current position by applying the speed estimate to the
424 // most recent transition
425 position = prev_trans_pos + (((int64_t)speed_2_20 * (step_us - transition_us)) >> 20);
426
427 // make sure the position estimate is between "low" and "high", as we
428 // can be sure the actual current position must be in this range
429 if ((int)(position - high) > 0)
430 position = high;
431 else if ((int)(position - low) < 0)
432 position = low;
433 }
434
435 // save the current values to use on the next sample
436 prev_low = low;
437 prev_high = high;
438 step = new_step;
439 prev_step_us = step_us;
440}
